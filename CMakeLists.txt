lfl_clear_projects()
set(LFL_PNG 1)
set(LFL_GIF 1)
set(LFL_JPEG 1)
set(LFL_HEADLESS 1)
if(LFL_ANDROID)
elseif(LFL_IPHONE)
else(LFL_ANDROID)
  set(LFL_GLOG 1)
endif(LFL_ANDROID)

lfl_project(lterm-sandbox-render)
add_subdirectory(../core/lfapp ${CMAKE_CURRENT_BINARY_DIR}/${LFL_PROJECT}_lfapp_obj)
add_executable(lterm-sandbox-render render.cpp)
target_link_libraries(lterm-sandbox-render ${LFAPP_LIB})
target_compile_definitions(lterm-sandbox-render PUBLIC ${LFAPP_DEF})
target_include_directories(lterm-sandbox-render PUBLIC ${LFAPP_INCLUDE})
lfapp_post_build_copy_bin(lterm-sandbox-render lterm-sandbox-render LTerminal)

lfl_clear_projects()
set(LFL_PNG 1)
set(LFL_GLES2 1)
set(LFL_OPENSSL 1)
set(LFL_LFLVIDEO 1)
set(LFL_LFLINPUT 1)
if(NOT WIN32)
  set(LFL_ICONV 1)
endif()
if(LFL_ANDROID)
elseif(LFL_IPHONE)
else()
  set(LFL_GLEW 1)
  if(NOT WIN32)
    set(LFL_GLOG 1)
    set(LFL_FREETYPE 1)
  endif()
endif()

lfl_project(lterm)
add_subdirectory(../core/lfapp ${CMAKE_CURRENT_BINARY_DIR}/${LFL_PROJECT}_lfapp_obj)
lfapp_add_target(lterm term.cpp term.rc)
target_compile_definitions(lterm PUBLIC ${LFAPP_DEF})
target_include_directories(lterm PUBLIC ${LFAPP_INCLUDE})
lfapp_post_build_start(lterm lterm LTerminal)
add_dependencies(lterm-sandbox-render lterm)
